#include <bits/stdc++.h>
using namespace std;
#define fastio                 \
  ios::sync_with_stdio(false); \
  cin.tie(0);
#define endl '\n'

#define int long long

int n, m;
vector<int> netswitch, cost;
vector<int> dp;

int32_t main(void)
{
  fastio;

  cin >> n;
  netswitch.resize(n);
  cost.resize(n);

  for (int i = 0; i < n; i++)
    cin >> netswitch[i] >> cost[i];

  cin >> m;

  if (m == 1)
    cout << 0 << endl;
  else
  {
    dp.resize(m + 1, LLONG_MAX);

    // dp[n] = nê°œì˜ ì»´í“¨í„°ë¥¼ ì„¤ì¹˜í•  ë•Œ ìµœì†Œ ë¹„ìš©
    dp[0] = 0;

    for (int i = 0; i < n; i++)
    {
      // í•˜ë‚˜ëŠ” ì¸í„°ë„· í•˜ë‚˜ëŠ” ë‹¤ë¥¸ ê³µìœ ê¸° ë‚˜ë¨¸ì§€ ì»´í“¨í„°ë¼ê³  ìƒê°
      for (int j = m - (netswitch[i] - 2); j >= 1; j--)
      {
        // ê¹”ë”í•œ ì—°ê²°ì´ ì•„ë‹˜
        if (dp[j] == LLONG_MAX)
          continue;

        // ìµœì†Œ ë¹„ìš© ì—°ê²°
        dp[j + (netswitch[i] - 2)] = min(dp[j + (netswitch[i] - 2)], dp[j] + cost[i]);
      }

      // ë‹¤ë¥¸ ìŠ¤ìœ„ì¹˜ë¥¼ ê¼½ì§€ ì•Šê³  ì¸í„°ë„·ê³¼ ì»´í“¨í„°ë§Œ ì—°ê²°ë˜ì–´ ìˆëŠ” ê²½ìš°
      if (netswitch[i] - 1 <= m)
        dp[netswitch[i] - 1] = min(dp[netswitch[i] - 1], cost[i]);
    }

    cout << (dp[m] == LLONG_MAX ? -1 : dp[m]) << endl;
  }

  return 0;
}
